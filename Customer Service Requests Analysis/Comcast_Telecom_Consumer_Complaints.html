# Function to update the Date-Month-year to Month-Year
def month_update(date):
    return_date = ''
    date_split = date.split('-')
    if len(date_split) > 2:
        return_date = date_split[1] + '-' + date_split[2]
    return return_date

# Function to update complaint type based on the string in the complaints
def complaint_type_update(complaint):
    if 'network' in complaint.lower():
        return 'Network'
    elif 'bill' in complaint.lower():
        return 'Billing'
    elif 'internet' in complaint.lower():
        return 'Internet'
    elif 'speed' in complaint.lower():
        return 'Speed'
    else:
        return 'Others'

# Function to set Solved and Closed to Closed, Open and Pending to Open
def status_update(status):
    if status.lower() in ['open', 'pending']:
        return 'Open'
    elif status.lower() in ['solved', 'closed']:
        return 'Closed'
